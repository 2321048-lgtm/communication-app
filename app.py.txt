import streamlit as st
import pandas as pd

st.set_page_config(page_title="의사소통기능 분류 도구", page_icon="💬")

st.title("💬 의사소통기능 분류 도구")
st.write("아동의 발화를 입력하면 의사소통기능을 간단히 분류해주는 웹앱입니다.")

st.markdown("---")

utterance = st.text_input("아동 발화를 입력하세요", placeholder="예: 물 주세요, 이거 뭐야?, 싫어")

def classify_function(text):
    text = text.strip()

    if text == "":
        return "입력 없음", "발화를 입력해주세요."

    # 요구하기
    request_keywords = ["주세요", "줘", "해줘", "도와줘", "하고 싶어", "먹고 싶어", "열어", "꺼내"]
    # 정보 요구하기
    question_keywords = ["뭐", "어디", "누구", "왜", "언제", "어떻게", "?"]
    # 거부/부정
    reject_keywords = ["싫어", "안 해", "안 할래", "아니", "그만", "하지마"]
    # 감정/상태 표현
    emotion_keywords = ["좋아", "무서워", "아파", "졸려", "배고파", "화나", "기뻐", "속상해"]
    # 진술하기
    statement_keywords = ["있어", "없어", "했어", "갔어", "먹었어", "봤어", "커", "작아"]

    if any(word in text for word in request_keywords):
        return "요구하기", "원하는 물건, 행동, 도움을 요청하는 발화입니다."

    elif any(word in text for word in question_keywords):
        return "정보 요구하기", "상대방에게 정보를 묻는 발화입니다."

    elif any(word in text for word in reject_keywords):
        return "거부/부정하기", "원하지 않거나 거절하는 의사소통 기능입니다."

    elif any(word in text for word in emotion_keywords):
        return "감정 및 상태 표현하기", "자신의 감정이나 신체 상태를 표현하는 발화입니다."

    elif any(word in text for word in statement_keywords):
        return "진술하기", "사물, 상황, 경험 등에 대해 말하는 발화입니다."

    else:
        return "기타", "정확한 분류가 어려운 발화입니다. 상황 맥락을 함께 확인하는 것이 좋습니다."

if st.button("분류하기"):
    category, explanation = classify_function(utterance)

    st.subheader("분류 결과")
    st.success(f"의사소통기능: {category}")
    st.write(explanation)

st.markdown("---")

st.subheader("예시 발화")

example_data = {
    "발화": [
        "물 주세요",
        "이거 뭐야?",
        "싫어",
        "배고파",
        "자동차 있어"
    ],
    "예상 의사소통기능": [
        "요구하기",
        "정보 요구하기",
        "거부/부정하기",
        "감정 및 상태 표현하기",
        "진술하기"
    ]
}

df = pd.DataFrame(example_data)
st.table(df)

st.caption("※ 본 도구는 키워드 기반의 간단한 분류 도구이므로 실제 평가에서는 상황 맥락과 비언어적 행동을 함께 고려해야 합니다.")
